//[문제1]
SELECT
d.department_name name, MIN(e.salary) minsal, MAX(e.salary) maxsal, AVG(e.salary) avgsal 
FROM departments d, employees e
WHERE d.department_id = e.department_id
GROUP BY d.department_name
ORDER BY name;

//[문제2]
SELECT
e.employee_id, e.department_id, e.salary, vt.avgs
FROM employees e, (SELECT
								 e.department_id id, round(avg(e.salary)) avgs
								 FROM departments d, employees e
								 WHERE d.department_id = e.department_id
								 GROUP BY e.department_id) vt
WHERE e.salary > vt.avgs AND
      e.department_id = vt.id;

//부서별 평균
SELECT
e.department_id, avg(e.salary) avgs
FROM departments d, employees e
WHERE d.department_id = e.department_id
GROUP BY e.department_id;
