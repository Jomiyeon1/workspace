SELECT
e1.first_name 사원이름, e1.hire_date, e.employee_id, e.first_name 매니저이름
FROM employees e, employees e1
WHERE e.employee_id(+) = e1.manager_id AND
	  e1.hire_date BETWEEN '1980-01-01' AND '1989-12-31';

// sales 부서 직원
SELECT
e.first_name, e.salary, d.department_name
FROM employees e, departments d, (SELECT
								  avg(salary) avgs
								  FROM employees
								  WHERE department_id = 100) vt
WHERE e.department_id = d.department_id AND
      e.department_id = (SELECT
						  department_id
						  FROM departments
						  WHERE department_name = 'Sales') AND
	  e.salary < vt.avgs;

// 100번부서의 평균급여
SELECT
avg(salary)
FROM employees
WHERE department_id = 100;
